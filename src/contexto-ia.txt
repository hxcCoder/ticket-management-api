// src/controllers/ticket.controller.ts
import { Request, Response, NextFunction } from &apos;express&apos;;
import { TicketService } from &apos;../services/ticket.service&apos;;

export class TicketController {
/**
   * Maneja la petición POST para crear un ticket
   */
static async createTicket(req: Request, res: Response, next: NextFunction): Promise&lt;void&gt; {
    try {
      // 1. Extraemos el ID del usuario directamente del token (¡NUNCA del body por seguridad!)
    const authorId = req.user!.id; 
    
      // 2. Extraemos los datos del body (que ya vienen limpios y validados por Zod)
    const { title, description, priority } = req.body;

      // 3. Delegamos la lógica de negocio al Servicio
    const ticket = await TicketService.createTicket({
        title,
        description,
        priority,
        authorId,
    });

      // 4. Retornamos la respuesta con el código HTTP correcto (201 Created)
    res.status(201).json({
        status: &apos;success&apos;,
        data: { ticket },
    });
    } catch (error) {
      // Si algo falla en la base de datos o el servicio, se lo pasamos al manejador global
        next(error); 
    }
    
}
static async getTickets(req: Request, res: Response, next: NextFunction): Promise&lt;void&gt; {
    try {
      // Extraemos ID y Rol del token de forma segura
      const userId = req.user!.id;
      const userRole = req.user!.role;

      // Llamamos al servicio pasando la información del usuario
      const tickets = await TicketService.getTickets(userId, userRole);

      res.status(200).json({
        status: &apos;success&apos;,
        results: tickets.length,
        data: { tickets },
      });
    } catch (error) {
      next(error);
    }
  }
}

// src/middlewares/auth.middleware.ts

import { Request, Response, NextFunction } from &apos;express&apos;;
import jwt from &apos;jsonwebtoken&apos;;
import { AppError } from &apos;./errorHandler&apos;; // Ajusta la ruta a tu AppError
import { TokenPayload, Role } from &apos;../types/express/index&apos;; // Ajusta la ruta

/**
 * Middleware de Autenticación (isAuth)
 * Verifica que el token exista y sea válido.
 */
export const isAuth = (req: Request, _res: Response, next: NextFunction) =&gt; {
    try {
    // 1. Extraer el token del header &quot;Authorization: Bearer &lt;token&gt;&quot;
    const authHeader = req.headers.authorization;
    
    if (!authHeader || !authHeader.startsWith(&apos;Bearer &apos;)) {
        throw new AppError(&apos;No estás autenticado. Por favor, inicia sesión.&apos;, 401);
    }

    const token = authHeader.split(&apos; &apos;)[1];

    // 2. Verificar el token
    // Ojo: En producción asegura tener process.env.JWT_SECRET definido
    const secret = process.env.JWT_SECRET || &apos;super-secret-dev-key&apos;; 
    const decoded = jwt.verify(token, secret) as TokenPayload;

    // 3. Inyectar el payload en la request para uso futuro
    req.user = decoded;

    next();
} catch (error) {
    if (error instanceof jwt.TokenExpiredError) {
        return next(new AppError(&apos;Tu sesión ha expirado. Inicia sesión nuevamente.&apos;, 401));
    }
    if (error instanceof jwt.JsonWebTokenError) {
        return next(new AppError(&apos;Token inválido.&apos;, 401));
    }
    next(error);
}
};

/**
 * Middleware de Autorización (hasRole)
 * Verifica que el usuario autenticado tenga los permisos necesarios.
 * Se debe usar SIEMPRE después de `isAuth`.
 */
export const hasRole = (roles: Role[]) =&gt; {
    return (req: Request, _res: Response, next: NextFunction) =&gt; {
    // req.user debería existir porque isAuth se ejecuta antes
    if (!req.user) {
        return next(new AppError(&apos;No se encontró información del usuario.&apos;, 401));
    }

    if (!roles.includes(req.user.role)) {
        return next(new AppError(&apos;No tienes permisos para realizar esta acción.&apos;, 403));
    }

    next();
};
};

import { Request, Response, NextFunction } from &apos;express&apos;;
import { ZodError } from &apos;zod&apos;;

// 1. Clase base para todos nuestros errores operacionales
export class AppError extends Error {
    public readonly statusCode: number;

constructor(message: string, statusCode: number) {
    super(message);
    this.statusCode = statusCode;
    
    // FIX: Restaura la cadena de prototipos correctamente
    Object.setPrototypeOf(this, new.target.prototype);
}
}

// 2. Errores comunes predefinidos
export class NotFoundError extends AppError {
    constructor(message = &apos;Resource not found&apos;) {
    super(message, 404);
}
}

export class BadRequestError extends AppError {
    constructor(message = &apos;Bad request&apos;) {
    super(message, 400);
}
}

export class UnauthorizedError extends AppError {
    constructor(message = &apos;Unauthorized&apos;) {
    super(message, 401);  }
}

export class ForbiddenError extends AppError {
  constructor(message = &apos;Forbidden&apos;) {
    super(message, 403);
  }
}

// 3. Middleware Global
export const errorHandler = (
    err: Error,
    _req: Request,
    res: Response,
    _next: NextFunction
): void =&gt; {
  // Caso A: Validación de Zod
if (err instanceof ZodError) {
    res.status(400).json({
        status: &apos;error&apos;,
        message: &apos;Validation failed&apos;,
      // FIX: Usamos issues y tipamos el parámetro inferido
        errors: err.issues.map((e) =&gt; ({
        field: e.path.join(&apos;.&apos;),
        message: e.message,
        })),
    });
    return;
}

  // Caso B: Errores controlados
if (err instanceof AppError) {
    res.status(err.statusCode).json({
        status: &apos;error&apos;,
        message: err.message,
    });
    return;
}

  // Caso C: Errores inesperados
console.error(&apos; Unexpected Error:&apos;, err);

res.status(500).json({
    status: &apos;error&apos;,
    message: process.env.NODE_ENV === &apos;production&apos; 
        ? &apos;Internal server error&apos; 
        : err.message,
    });
};

// src/middlewares/validateRequest.ts
import { Request, Response, NextFunction } from &apos;express&apos;;
import { ZodObject } from &apos;zod&apos;;

export const validateRequest = (schema: ZodObject&lt;any&gt;) =&gt; {
    return async (req: Request, _res: Response, next: NextFunction): Promise&lt;void&gt; =&gt; {
    try {
      // Valida body, query y params al mismo tiempo según el esquema provisto
    await schema.parseAsync({
        body: req.body,
        query: req.query,
        params: req.params,
    });
    next();
    } catch (error) {
      // Envía el error de Zod directamente al errorHandler global
    next(error);
    }
};
};

import { Router } from &apos;express&apos;;
import { TicketController } from &apos;../controllers/ticket.controller&apos;;

    TicketController.getTickets
);

import { prisma } from &apos;../config/prisma&apos;;
import { Role, TicketPriority } from &apos;@prisma/client&apos;;

interface CreateTicketDTO {
    title: string;
    description: string;
    priority?: TicketPriority;
    authorId: string;
}

export class TicketService {
    /**
     * Crea un nuevo ticket en el sistema
     */
    static async createTicket(data: CreateTicketDTO) {
        const ticket = await prisma.ticket.create({
            data: {
                title: data.title,
                description: data.description,
                priority: data.priority,
                authorId: data.authorId
            },
            select: {
                id: true,
                title: true,
                status: true,
                priority: true,
                createdAt: true,
                author: {
                    select: {
                        id: true,
                        name: true,
                        email: true
                    }
                }
            }
        });
        return ticket;
    }

    static async getTickets(userId: string, userRole: Role) {
        let whereCondition = {};

        if (userRole === &apos;CUSTOMER&apos;) {
            whereCondition = { authorId: userId };
        }

        const tickets = await prisma.ticket.findMany({
            where: whereCondition,
            orderBy: {
                createdAt: &apos;desc&apos;,
            },
            select: {
                id: true,
                title: true,
                status: true,
                priority: true,
                createdAt: true,
                author: {
                    select: { name: true, email: true }
                }
            }
        });

        return tickets;
    }
}
